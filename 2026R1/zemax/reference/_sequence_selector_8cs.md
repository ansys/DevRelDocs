        public override string ToString()
        {
            if (HasDiffractionX && HasDiffractionY)
                return string.Format(
                    "{0}.{1}/({2},{3})",
                    ObjectNumber,
                    FaceNumber,
                    DiffOrderX,
                    DiffOrderY
                );

                    ObjectNumber,
                    FaceNumber,
                    DiffOrderX
                );
            if (HasDiffractionY)
                return string.Format(
                    "{0}.{1}/(*,{2})",
                    ObjectNumber,
                    FaceNumber,
                    DiffOrderY
                );
            return string.Format("{0}.{1}", ObjectNumber, FaceNumber);
        }
