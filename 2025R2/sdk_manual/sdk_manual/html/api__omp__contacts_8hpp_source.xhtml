49template <typename ContactFunctions>
50void dispatch_frictional_contact_forces_kernel(
51 IRockyModel& model, const ContactFunctions& function_ptrs, void* udf_data)
52{
53#ifndef GPU_KERNEL_COMPILING_STAGE
54 SDeviceModel* device_data = model.get_device_data(0);
55
56 const auto n_contacts = device_data->n_contacts;
57 if (n_contacts->particle_fr == 0 && n_contacts->triangle_fr == 0)
58 return;
59
60 #pragma omp parallel num_threads(model.get_number_of_processes())
61 {
62 // particles -> particles
63 device::api_compute_particle_contact_frictional_forces<OMPBackend>(
64 device_data, model.get_current_time(), function_ptrs, udf_data);
65
66 // particles -> triangles
67 device::api_compute_triangle_contact_frictional_forces<OMPBackend>(
68 device_data, model.get_current_time(), function_ptrs, udf_data);
69 }
70#endif
71}
