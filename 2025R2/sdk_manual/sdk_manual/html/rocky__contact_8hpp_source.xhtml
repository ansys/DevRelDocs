443 inline ROCKY_FUNCTIONS double get_home_rebound_coefficient() const
444 {
445 // At discrete breakage when a joint is fully broken there could be significant overlap
446 // of the fragments initially if they are at high stress. The force on this overlap is
447 // supposed to be zero, but it can jump to quite high value if coeff_r is large.
448 // Then, we reduce the rebound coefficient to a value near zero to avoid these fluctuations.
449 if (this->get_home_particle_group().is_breakage_discrete() && this->is_particle_contact
450 && this->model->get_particle_geometry(this->get_home_particle_index()).dgb_parent_index
451 == this->model->get_particle_geometry(this->get_near_particle_index())
452 .dgb_parent_index)
453 {
454 return ROCKY_SMALL_VALUE;
455 }
456 return this->get_home_particle_group().coeff_r;
457 }
458
459 inline ROCKY_FUNCTIONS double get_near_rebound_coefficient() const
460 {
461 // At discrete breakage when a joint is fully broken there could be significant overlap
462 // of the fragments initially if they are at high stress. The force on this overlap is
463 // supposed to be zero, but it can jump to quite high value if coeff_r is large.
464 // Then, we reduce the rebound coefficient to a value near zero to avoid these fluctuations.
465 if (this->is_particle_contact)
466 {
467 if (this->get_near_particle_group().is_breakage_discrete()
468 && this->model->get_particle_geometry(this->get_home_particle_index())
469 .dgb_parent_index
470 == this->model->get_particle_geometry(this->get_near_particle_index())
471 .dgb_parent_index)
472 {
473 return ROCKY_SMALL_VALUE;
474 }
475 return this->get_near_particle_group().coeff_r;
476 }
477 return this->get_near_particle_group().coeff_r;
478 }

482 inline ROCKY_FUNCTIONS double get_equivalent_stiffness() const
483 {
484 if (this->get_normal_force_model() != nfmHertzViscous)
485 {
486 double home_stiffness = this->get_home_stiffness();
487 double near_stiffness = this->get_near_stiffness();
488 return (home_stiffness * near_stiffness) / (home_stiffness + near_stiffness);
489 }
490 else // TODO remove this ROCKY-13106
491 {
492 const double sqrt_overlap = sqrt(max(0.0, this->get_overlap()));
493 const double home_size = this->get_home_particle().get_original_size();
494 if (this->is_particle_contact)
495 {
496 const double near_size = this->get_near_particle().get_original_size();
497 return (4.0 / 3.0) * this->get_reduced_young_modulus()
498 * sqrt(0.5 * home_size * near_size / (home_size + near_size)) * sqrt_overlap;
499 }
500 else
501 return (4.0 / 3.0) * this->get_reduced_young_modulus() * sqrt(0.5 * home_size)
502 * sqrt_overlap;
503 }
504 }

510 return (home_reduced_ym * near_reduced_ym) / (home_reduced_ym + near_reduced_ym);
511 }

540 For more details, please refer to the DEM Technical Manual. The value of this damping ratio

542 described in the referred manual, to the Hertzian Spring Dashpot normal force
543 contact models.

550 /*
551 Overlap time derivative for models that include viscous damping. Because of stability reasons,
552 for concave particles the time derivative is approximated using the normal relative velocity.
553 */
554 inline ROCKY_FUNCTIONS double get_overlap_time_derivative(
555 const double normal_relative_velocity) const
556 {
557 double previous_overlap = this->get_previous_overlap();
558 const double overlap = this->get_overlap();
559 const double timestep = this->get_timestep();
560 if (this->model->gen->flags.has_concave_particles()
561 && (this->get_state().is_home_simple_composite()
562 || (this->is_particle_contact && this->get_state().is_near_simple_composite())))
563 {
564 previous_overlap = overlap + normal_relative_velocity * timestep;
565 }
566 return (overlap - rocky::dmax(previous_overlap, 0.0)) / timestep;
567 }

569 inline ROCKY_FUNCTIONS double get_tangential_stiffness(
570 cuda_contact_intermediate_data& idata) const
571 {
572 const auto mi = this->get_material_interaction();
573 const auto normal_model = this->get_normal_force_model();
574
575 if (normal_model == nfmLinearHysteresis || normal_model == nfmLinearElasticViscous)
576 {
577 return idata.stiffness * mi.kt_kn;
578 }
579 else if (normal_model == ENormalForceModel::nfmHertzViscous)
580 {
581 return 1.5 * idata.stiffness * mi.kt_kn;
582 }
583 else if (this->get_normal_force_model() == nfmCustom)
584 {
585 return this->get_equivalent_stiffness() * mi.kt_kn;
586 }
587 else
588 {
589 assert(0);
590 return 0;
591 }
592 }

594 inline ROCKY_FUNCTIONS double get_cgm_average_factor() const

596 const double home_factor = this->get_home_particle_base_group().cgm_scale_factor;

599 return home_factor;
